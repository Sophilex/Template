inline int read()
{
	int X=0; bool flag=1; char ch=getchar();
	while(ch<'0'||ch>'9') {if(ch=='-') flag=0; ch=getchar();}
	while(ch>='0'&&ch<='9') {X=(X<<1)+(X<<3)+ch-'0'; ch=getchar();}
	if(flag) return X;
	return ~(X-1);
}

## ksc

```
inline ll ksc(ll x,ll y,ll p){//计算x乘y的积
	ll res=0;//加法初始化
	while(y){
		if(y&1)res=(res+x)%p;//模仿二进制
		x=(x<<1)%p; y>>=1;//将x不断乘2达到二进制
	}
	return res;
}
```
